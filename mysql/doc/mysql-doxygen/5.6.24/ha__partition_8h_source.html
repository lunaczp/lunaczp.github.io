50  num_parts= 0;

52  }
53  ~Parts_share_refs()
54  {
55  uint i;
56  for (i= 0; i < num_parts; i++)
57  if (ha_shares[i])
58  delete ha_shares[i];
59  if (ha_shares)
60  delete [] ha_shares;
61  }
62  bool init(uint arg_num_parts)
63  {
64  DBUG_ASSERT(!num_parts && !ha_shares);
65  num_parts= arg_num_parts;
66  /* Allocate an array of Handler_share pointers */
67  ha_shares= new Handler_share *[num_parts];
68  if (!ha_shares)
69  {
70  num_parts= 0;
71  return true;
72  }
73  memset(ha_shares, 0, sizeof(Handler_share*) * num_parts);
74  return false;
75  }

507  -------------------------------------------------------------------------
508  MODULE full table scan
509  -------------------------------------------------------------------------
510  This module is used for the most basic access method for any table
511  handler. This is to fetch all data through a full table scan. No
512  indexes are needed to implement this part.
513  It contains one method to start the scan (rnd_init) that can also be
514  called multiple times (typical in a nested loop join). Then proceeding
515  to the next record (rnd_next) and closing the scan (rnd_end).
516  To remember a record for later access there is a method (position)
517  and there is a method used to retrieve the record based on the stored
518  position.
519  The position can be a file position, a primary key, a ROWID dependent
520  on the handler below.
521  -------------------------------------------------------------------------
522  */

524  unlike index_init(), rnd_init() can be called two times
525  without rnd_end() in between (it only makes sense if scan=1).
526  then the second call should prepare for the new table scan
527  (e.g if rnd_init allocates the cursor, second call should
528  position it to the start of the table, no need to deallocate
529  and allocate it again
530  */

538  /*
539  -------------------------------------------------------------------------
540  MODULE index scan
541  -------------------------------------------------------------------------
542  This part of the handler interface is used to perform access through
543  indexes. The interface is defined as a scan interface but the handler
544  can also use key lookup if the index is a unique index or a primary
545  key index.
546  Index scans are mostly useful for SELECT queries but are an important
547  part also of UPDATE, DELETE, REPLACE and CREATE TABLE table AS SELECT
548  and so forth.
549  Naturally an index is needed for an index scan and indexes can either
550  be ordered, hash based. Some ordered indexes can return data in order
551  but not necessarily all of them.
552  There are many flags that define the behavior of indexes in the
553  various handlers. These methods are found in the optimizer module.
554  -------------------------------------------------------------------------
555 
556  index_read is called to start a scan of an index. The find_flag defines
557  the semantics of the scan. These flags are defined in
558  include/my_base.h
559  index_read_idx is the same but also initializes index before calling doing
560  the same thing as index_read. Thus it is similar to index_init followed
561  by index_read. This is also how we implement it.
562 
563  index_read/index_read_idx does also return the first row. Thus for
564  key lookups, the index_read will be the only call to the handler in
565  the index scan.
566 
567  index_init initializes an index before using it and index_end does
568  any end processing needed.
569  */

571  key_part_map keypart_map,
