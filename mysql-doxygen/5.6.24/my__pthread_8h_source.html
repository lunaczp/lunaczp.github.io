278 /*
279  We define my_sigset() and use that instead of the system sigset() so that
280  we can favor an implementation based on sigaction(). On some systems, such
281  as Mac OS X, sigset() results in flags such as SA_RESTART being set, and
282  we want to make sure that no such flags are set.
283 */
284 #if defined(HAVE_SIGACTION) && !defined(my_sigset)
285 #define my_sigset(A,B) do { struct sigaction l_s; sigset_t l_set; \
286  DBUG_ASSERT((A) != 0); \
287  sigemptyset(&l_set); \
288  l_s.sa_handler = (B); \
289  l_s.sa_mask = l_set; \
290  l_s.sa_flags = 0; \
291  sigaction((A), &l_s, NULL); \
292  } while (0)
293 #elif defined(HAVE_SIGSET) && !defined(my_sigset)
294 #define my_sigset(A,B) sigset((A),(B))
295 #elif !defined(my_sigset)
296 #define my_sigset(A,B) signal((A),(B))
