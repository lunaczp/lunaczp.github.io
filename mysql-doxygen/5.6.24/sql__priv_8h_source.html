177 /*
178  Check how many bytes are available on buffer.
179 
180  @param buf_start Pointer to buffer start.
181  @param buf_current Pointer to the current position on buffer.
182  @param buf_len Buffer length.
183 
184  @return Number of bytes available on event buffer.
185 */
186 template <class T> T available_buffer(const char* buf_start,
187  const char* buf_current,
188  T buf_len)
189 {
190  return buf_len - (buf_current - buf_start);
191 }
192 
193 /*
194  Check if jump value is within buffer limits.
195 
196  @param jump Number of positions we want to advance.
197  @param buf_start Pointer to buffer start
198  @param buf_current Pointer to the current position on buffer.
199  @param buf_len Buffer length.
200 
201  @return True If jump value is within buffer limits.
202  False Otherwise.
203 */
204 template <class T> bool valid_buffer_range(T jump,
205  const char* buf_start,
206  const char* buf_current,
207  T buf_len)
208 {
209  return (jump <= available_buffer(buf_start, buf_current, buf_len));
210 }
