55  TL_WRITE_ALLOW_WRITE,

57  WRITE lock used by concurrent insert. Will allow

60  TL_WRITE_CONCURRENT_INSERT,
61  /* Write used by INSERT DELAYED. Allows READ locks */
62  TL_WRITE_DELAYED,

64  parser only! Late bound low_priority flag.

66  */
67  TL_WRITE_DEFAULT,

71  TL_WRITE,
